phy.init({})

phy.init({ [param:String type], [param:Boolean worker], [param:Function callback], [param:Object3D scene]}, [param:string path]} )

type — the physics engine type
worker — run in worker or not
callback — function start after physics engine init
scene — the three.js main scene
path — set default resource path
